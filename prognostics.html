#taking readings from 3 sensors, one should be constant at 5 volt, one should be around 3 volt with noise, one should be a sine wave

import numpy as np
readings = []

for i in np.linspace(0, 2*np.pi, 100): #take 100 values between 0 and 2pi
    reading_1 = 5 #constant at 5v
    reading_2 = 3 + np.random.normal(scale=0.25) #around 3v with added Gaussian noise w/ std of 0.25
    reading_3 = np.sin(i) #sine wave
    readings.append([reading_1, reading_2, reading_3]) #each reading has 3 data points

#now simulate a fault with the readings, first and last quarters are fine
#the middle half has the 5v constant rise to 10v

readings_fault = []

for i in np.linspace(0, 2*np.pi, 100):
    if i < 0.5*np.pi or i > 1.5*np.pi: #if in the first and last quarters, take normal readings
        reading_1 = 5
        reading_2 = 3 + np.random.normal(scale=0.25)
        reading_3 = np.sin(i)
        readings_fault.append([reading_1, reading_2, reading_3])
    else: #else, reading_1 now equals 10
        reading_1 = 10
        reading_2 = 3 + np.random.normal(scale=0.25)
        reading_3 = np.sin(i)
        readings_fault.append([reading_1, reading_2, reading_3])

#Gaussian Mixture models are a way of automatically clustering data
#downside is you have to know how many clusters are required in advance

from sklearn.mixture import GaussianMixture

gmm = GaussianMixture(2) #state the number of clusters, here we have 2, faulty and not-faulty
gmm.fit(X) #fit to our data
labels = gmm.predict(X) #get a list with the label for each data point in X
